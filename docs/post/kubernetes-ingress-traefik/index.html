apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: ingress-traefik-test
  # ingressのnamespaceとserviceのnamespaceは同じにする
  #namespace: ns-traefik

spec:
  rules:
    # hostを設定しないとIPアドレスでアクセスできる
    #- host: ingress1.internal
    - http:
        paths:
          - path: /
            pathType: Prefix
            backend:
              service:
                name: nginx-deployment-lb
                port:
                  number: 83
    - host: ingress1.internal
      http:
        paths:
          - path: /
            pathType: Prefix
            backend:
              service:
                name: nginx-deployment-lb
                port:
                  number: 83
    - host: ingress2.internal
      http:
        paths:
          - path: /
            pathType: Prefix
            backend:
              service:
                name: nginx-deployment-lb
                port:
                  number: 83
---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx-deployment
  #namespace: ns-traefik
  labels:
    app: nginx
spec:
  replicas: 10
  selector:
    matchLabels:
      app: nginx
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
      - name: nginx
        image: nginx:1.27
        ports:
        - containerPort: 80
---
apiVersion: v1
kind: Service
metadata:
  name: nginx-deployment-lb
  #namespace: ns-traefik
spec:
  type: NodePort
  ports:
  - port: 83
    targetPort: 80
  selector:
    app: nginx

デプロイする

1
2

kubectl apply -f traefik-ingress.yaml
kubectl get ingress -A

アクセスできるか確認する

IPアドレスとバーチャルホストでアクセスできるか確認する
結果が帰ってくればOK
アクセスできない場合はportがあっているかnamespaceが同じか等確認をする

1
2
3

curl http://192.168.10.55
curl -H 'Host:ingress1.internal' http://192.168.10.55
curl -H 'Host:ingress2.internal' http://192.168.10.55

削除する

1
2

kubectl delete -f traefik-ingress.yaml
helm uninstall traefik -n ns-traefik

1
2

kubectl get services -A
helm ls -A
